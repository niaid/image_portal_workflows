29def SLURM_exec(asynchronous: bool = False, **cluster_kwargs): 

30 """ 

31 brings up a dynamically sized cluster. 

32 

33 Docs: https://jobqueue.dask.org/en/latest/generated/dask_jobqueue.SLURMCluster.html 

34 

35 We can view the sbatch script using the following command, to know how the job is started 

36 by slurm: 

37 python -c "from em_workflows import config; c = config.SLURM_exec(); print(c.job_script())" 

38 

39 The processes determins number of dask workers, and nthreads = cores / processes 

40 The memory limit is also divided among the workers 

41 

42 More about the cluster: https://bigskywiki.niaid.nih.gov/big-sky-architecture 

43 """ 

44 home = os.environ["HOME"] 

45 env_name = os.environ["HEDWIG_ENV"] 

46 flowrun_id = os.environ.get("PREFECT__FLOW_RUN_ID", "not-found") 

47 job_script_prologue = [ 

48 f"source /gs1/home/hedwig_{env_name}/{env_name}/bin/activate", 

49 "echo $PATH", 

50 ] 

51 cluster = SLURMCluster( 

52 name="dask-worker", 

53 # processes=4, 

54 death_timeout=121, 

55 local_directory=f"{home}/dask_tmp/", 

56 log_directory=f"{home}/slurm-log/{flowrun_id}", 

57 job_script_prologue=job_script_prologue, 

58 # queue is arg for SBATCH --partition 

59 # to learn more about partitions, run `sinfo` in hpc 

60 queue="int", 

61 walltime="4:00:00", 

62 # job_extra_directives=["--gres=gpu:1"], 

63 asynchronous=asynchronous, 

64 **cluster_kwargs, 

65 ) 

66 cluster.scale(1) 

67 # cluster.adapt(minimum=1, maximum=6) 

68 # to get logger, we must be within an active flow/task run 

69 print("Dask cluster started") 

70 print(f"see dashboard {cluster.dashboard_link}") 

71 return cluster
